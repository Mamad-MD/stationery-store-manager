import React, { useState, useEffect } from 'react';
import { Text, View, StyleSheet, Button, TextInput, Alert, ScrollView, FlatList, TouchableOpacity } from 'react-native';
import { CameraView, useCameraPermissions } from 'expo-camera';

export default function App() {
  const [permission, requestPermission] = useCameraPermissions();
  const [scanned, setScanned] = useState(false);
  const [mode, setMode] = useState('SCAN'); // MODES: SCAN, ADD_PRODUCT, REPORT
  const [currentBarcode, setCurrentBarcode] = useState('');
  const [products, setProducts] = useState({});
  const [totalSales, setTotalSales] = useState(0);

  if (!permission) {
    return <View style={styles.center}><Text>در حال بارگذاری...</Text></View>;
  }

  if (!permission.granted) {
    return (
      <View style={styles.center}>
        <Text style={{ textAlign: 'center', marginBottom: 20 }}>برای اسکن بارکد به دسترسی دوربین نیاز داریم</Text>
        <Button onPress={requestPermission} title="دادن دسترسی دوربین" />
      </View>
    );
  }

  const handleBarCodeScanned = ({ data }) => {
    setScanned(true);
    setCurrentBarcode(data);
    
    if (products[data]) {
      Alert.alert('محصول پیدا شد', `نام: ${products[data].name}\nقیمت: ${products[data].price}\nموجودی: ${products[data].stock}`, [
        { text: 'فروش (1 عدد)', onPress: () => sellProduct(data) },
        { text: 'ویرایش', onPress: () => setMode('ADD_PRODUCT') },
        { text: 'انصراف', onPress: () => setScanned(false) }
      ]);
    } else {
      setMode('ADD_PRODUCT');
    }
  };

  const sellProduct = (barcode) => {
    if (products[barcode].stock > 0) {
      const updated = { ...products };
      updated[barcode].stock -= 1;
      setProducts(updated);
      setTotalSales(prev => prev + parseInt(products[barcode].price));
      Alert.alert('✅ موفق', 'فروش ثبت شد و از موجودی کسر گردید.');
    } else {
      Alert.alert('❌ خطا', 'موجودی انبار صفر است!');
    }
    setScanned(false);
  };

  const addProduct = (name, price, stock) => {
    if(!name || !price || !stock) {
      Alert.alert("خطا", "لطفاً تمام فیلدها را پر کنید");
      return;
    }
    setProducts({ ...products, [currentBarcode]: { name, price, stock: parseInt(stock) } });
    setMode('SCAN');
    setScanned(false);
    Alert.alert("ذخیره شد", "محصول با موفقیت به انبار اضافه شد.");
  };

  return (
    <View style={styles.container}>
      {mode === 'SCAN' ? (
        <View style={{ flex: 1 }}>
          <CameraView
            onBarcodeScanned={scanned ? undefined : handleBarCodeScanned}
            barcodeScannerSettings={{ barcodeTypes: ["qr", "ean13", "code128"] }}
            style={StyleSheet.absoluteFillObject}
          />
          <View style={styles.uiHeader}>
            <View style={styles.badge}><Text style={styles.badgeText}>فروش: {totalSales}</Text></View>
            <TouchableOpacity style={styles.btnReport} onPress={() => setMode('REPORT')}>
               <Text style={{color: '#fff'}}>آمار انبار</Text>
            </TouchableOpacity>
          </View>
          {scanned && (
            <View style={styles.bottomBtn}>
               <Button title="اسکن مجدد" onPress={() => setScanned(false)} />
            </View>
          )}
        </View>
      ) : mode === 'ADD_PRODUCT' ? (
        <AddProductView onSave={addProduct} barcode={currentBarcode} onCancel={() => {setMode('SCAN'); setScanned(false);}} />
      ) : (
        <ReportView products={products} onBack={() => setMode('SCAN')} />
      )}
    </View>
  );
}

function AddProductView({ onSave, barcode, onCancel }) {
  const [name, setName] = useState('');
  const [price, setPrice] = useState('');
  const [stock, setStock] = useState('');

  return (
    <ScrollView style={styles.form}>
      <Text style={styles.title}>ثبت محصول | بارکد: {barcode}</Text>
      <TextInput placeholder="نام کالا (مثلاً: نوشابه)" style={styles.input} onChangeText={setName} />
      <TextInput placeholder="قیمت فروش (تومان)" keyboardType="numeric" style={styles.input} onChangeText={setPrice} />
      <TextInput placeholder="تعداد موجودی در انبار" keyboardType="numeric" style={styles.input} onChangeText={setStock} />
      <View style={{gap: 10}}>
        <Button title="ذخیره محصول" color="#2196F3" onPress={() => onSave(name, price, stock)} />
        <Button title="انصراف" color="#F44336" onPress={onCancel} />
      </View>
    </ScrollView>
  );
}

function ReportView({ products, onBack }) {
  const list = Object.entries(products);
  return (
    <View style={styles.form}>
      <Text style={styles.title}>📦 وضعیت موجودی انبار</Text>
      {list.length === 0 ? <Text style={{marginBottom: 20}}>هیچ کالایی ثبت نشده است.</Text> : 
      <FlatList 
        data={list} 
        keyExtractor={(item) => item[0]}
        renderItem={({ item }) => (
          <View style={styles.listItem}>
            <Text style={{fontWeight: 'bold'}}>{item[1].name}</Text>
            <Text>قیمت: {item[1].price} | موجودی: {item[1].stock} عدد</Text>
          </View>
        )} 
      />}
      <Button title="بازگشت به دوربین" onPress={onBack} />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#000' },
  center: { flex: 1, justifyContent: 'center', alignItems: 'center', padding: 20 },
  uiHeader: { position: 'absolute', top: 50, left: 20, right: 20, flexDirection: 'row', justifyContent: 'space-between' },
  badge: { backgroundColor: 'rgba(0,0,0,0.7)', padding: 10, borderRadius: 10, borderWidth: 1, borderColor: '#fff' },
  badgeText: { color: '#fff', fontSize: 16, fontWeight: 'bold' },
  btnReport: { backgroundColor: '#4CAF50', padding: 10, borderRadius: 10 },
  bottomBtn: { position: 'absolute', bottom: 50, left: 50, right: 50 },
  form: { flex: 1, backgroundColor: '#fff', padding: 30, paddingTop: 60 },
  title: { fontSize: 18, fontWeight: 'bold', marginBottom: 25, color: '#333', textAlign: 'center' },
  input: { backgroundColor: '#f9f9f9', padding: 15, marginBottom: 15, borderRadius: 10, borderWidth: 1, borderColor: '#eee', fontSize: 16 },
  listItem: { padding: 15, borderBottomWidth: 1, borderColor: '#eee', marginBottom: 5 }
});
